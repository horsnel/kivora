    75→    75→// ── Simple Markdown Renderer ──
    76→    76→function renderMarkdown(text, { skipDuplicateSources = false } = {}) {
    77→    77→  if (!text) return null
    78→    78→  const lines = text.split('\n')
    79→    79→  const elements = []
    80→    80→  let inList = false
    81→    81→  let listItems = []
    82→    82→  let inTable = false
    83→    83→  let tableRows = []
    84→    84→  let tableHeaders = []
    85→    85→  let keyIdx = 0
    86→    86→  let skipMode = false // When true, we're inside a duplicate Sources section — skip all content
    87→    87→
    88→    88→  function flushList() {
    89→    89→    if (listItems.length > 0) {
    90→    90→      elements.push(<ul key={`ul-${keyIdx++}`} className="space-y-1 mb-3 ml-1">{listItems}</ul>)
    91→    91→      listItems = []
    92→    92→    }
    93→    93→    inList = false
    94→    94→  }
    95→    95→
    96→    96→  function flushTable() {
    97→    97→    if (tableHeaders.length > 0) {
    98→    98→      elements.push(
    99→    99→        <div key={`tbl-wrap-${keyIdx++}`} className="-mx-1 px-1 mb-3">
   100→   100→          <table className="w-full border-collapse">
   101→   101→            <thead>
   102→   102→              <tr>{tableHeaders.map((h, hi) => (
   103→   103→                <th key={`th-${hi}`}>{h}</th>
   104→   104→              ))}</tr>
   105→   105→            </thead>
   106→   106→            <tbody>
   107→   107→              {tableRows.map((row, ri) => (
   108→   108→                <tr key={`tr-${ri}`}>{row.map((cell, ci) => (
   109→   109→                  <td key={`td-${ri}-${ci}`}>{renderInline(cell)}</td>
   110→   110→                ))}</tr>
   111→   111→              ))}
   112→   112→            </tbody>
   113→   113→          </table>
   114→   114→        </div>
   115→   115→      )
   116→   116→    }
   117→   117→    tableHeaders = []
   118→   118→    tableRows = []
   119→   119→    inTable = false
   120→   120→  }
   121→   121→
   122→   122→  function renderInline(str) {
   123→   123→    const parts = []
   124→   124→    const regex = /(\*\*|__)(.*?)\1|\[([^\]]+)\]\(([^)]+)\)|\[(\d+)\]/g
   125→   125→    let lastIndex = 0
   126→   126→    let match
   127→   127→    while ((match = regex.exec(str)) !== null) {
   128→   128→      if (match.index > lastIndex) {
   129→   129→        parts.push(str.slice(lastIndex, match.index))
   130→   130→      }
   131→   131→      if (match[1]) {
   132→   132→        parts.push(<strong key={`b-${keyIdx++}`} className="text-white font-semibold">{match[2]}</strong>)
   133→   133→      } else if (match[3] !== undefined && match[4] !== undefined) {
   134→   134→        parts.push(
   135→   135→          <a key={`a-${keyIdx++}`} href={match[4]} target="_blank" rel="noopener noreferrer" className="text-red-400 hover:text-red-300 underline underline-offset-2">
   136→   136→            {match[3]}
   137→   137→          </a>
   138→   138→        )
   139→   139→      } else if (match[5] !== undefined) {
   140→   140→        parts.push(<span key={`cit-${keyIdx++}`} className="citation-number">[{match[5]}]</span>)
   141→   141→      }
   142→   142→      lastIndex = match.index + match[0].length
   143→   143→    }
   144→   144→    if (lastIndex < str.length) parts.push(str.slice(lastIndex))
   145→   145→    return parts.length > 0 ? parts : str
   146→   146→  }
   147→   147→
   148→   148→  for (let i = 0; i < lines.length; i++) {
   149→   149→    const line = lines[i]
   150→   150→
   151→   151→    // Detect duplicate Sources/References heading — skip everything until the next heading of same or higher level
   152→   152→    if (skipDuplicateSources && DUPLICATE_SOURCE_HEADINGS.test(line.trim())) {
   153→   153→      flushList()
   154→   154→      flushTable()
   155→   155→      const headingLevel = line.match(/^(#{1,3})\s/)?.[1]?.length || 0
   156→   156→      skipMode = headingLevel
   157→   157→      continue
   158→   158→    }
   159→   159→    // Exit skip mode when we hit a new heading at same or higher level
   160→   160→    if (skipMode) {
   161→   161→      const newHeadingMatch = line.match(/^(#{1,3})\s/)
   162→   162→      if (newHeadingMatch) {
   163→   163→        const newLevel = newHeadingMatch[1].length
   164→   164→        if (newLevel <= skipMode) {
   165→   165→          skipMode = false
   166→   166→          // Don't continue — process this heading normally
   167→   167→        } else {
   168→   168→          continue // Still inside the duplicate section (sub-heading)
   169→   169→        }
   170→   170→      } else if (/^---+$/.test(line.trim())) {
   171→   171→        // Horizontal rule might end the section
   172→   172→        skipMode = false
   173→   173→        continue
   174→   174→      } else {
   175→   175→        continue // Skip content inside duplicate section
   176→   176→      }
   177→   177→    }
   178→   178→
   179→   179→    // Table row detection: lines containing pipes
   180→   180→    if (line.includes('|') && line.trim().startsWith('|')) {
   181→   181→      flushList()
   182→   182→      const cells = line.split('|').filter(c => c.trim() !== '').map(c => c.trim())
   183→   183→      // Separator row (|---|---|)
   184→   184→      if (cells.every(c => /^[-:]+$/.test(c))) {
   185→   185→        inTable = true
   186→   186→        continue
   187→   187→      }
   188→   188→      if (!inTable && tableHeaders.length === 0) {
   189→   189→        // First row = header
   190→   190→        tableHeaders = cells
   191→   191→        inTable = true
   192→   192→        continue
   193→   193→      }
   194→   194→      if (inTable) {
   195→   195→        tableRows.push(cells)
   196→   196→        continue
   197→   197→      }
   198→   198→    } else if (inTable) {
   199→   199→      // End of table
   200→   200→      flushTable()
   201→   201→    }
   202→   202→
   203→   203→    if (line.startsWith('### ')) { flushList(); elements.push(<h3 key={`h3-${keyIdx++}`} className="text-base font-semibold text-white mt-5 mb-2">{renderInline(line.slice(4))}</h3>); continue }
   204→   204→    if (line.startsWith('## ')) { flushList(); elements.push(<h2 key={`h2-${keyIdx++}`} className="text-lg font-semibold text-white mt-6 mb-2.5">{renderInline(line.slice(3))}</h2>); continue }
   205→   205→    if (line.startsWith('# ')) { flushList(); elements.push(<h1 key={`h1-${keyIdx++}`} className="text-xl font-bold text-white mt-4 mb-3">{renderInline(line.slice(2))}</h1>); continue }
   206→   206→    if (/^---+$/.test(line.trim())) { flushList(); elements.push(<hr key={`hr-${keyIdx++}`} className="border-[#1a1a1a] my-4" />); continue }
   207→   207→    if (/^[-*]\s/.test(line)) { inList = true; listItems.push(<li key={`li-${keyIdx++}`} className="text-sm text-[#a0a0a0] leading-relaxed pl-1">{renderInline(line.replace(/^[-*]\s/, ''))}</li>); continue }
   208→   208→    if (/^\d+\.\s/.test(line)) { inList = true; listItems.push(<li key={`li-${keyIdx++}`} className="text-sm text-[#a0a0a0] leading-relaxed pl-1 list-decimal ml-4">{renderInline(line.replace(/^\d+\.\s/, ''))}</li>); continue }
   209→   209→    if (!line.trim()) { flushList(); continue }
   210→   210→    flushList()
   211→   211→    elements.push(<p key={`p-${keyIdx++}`} className="text-sm text-[#a0a0a0] leading-relaxed mb-3">{renderInline(line)}</p>)
   212→   212→  }
   213→   213→  flushList()
   214→   214→  flushTable()
   215→   215→  return elements
   216→   216→}

   218→   218→// ── Filter duplicate Sources/References from HTML content (dangerouslySetInnerHTML path) ──
   219→   219→function filterDuplicateSources(html) {
   220→   220→  if (!html) return html
   221→   221→  // Remove sections headed by Sources/References/Bibliography etc.
   222→   222→  return html.replace(/<h[1-3][^>]*>\s*(Sources|References|Bibliography|Citations|Works?\s+Cited)\s*<\/h[1-3]>[\s\S]*?(?=<h[1-3][^>]*>|$)/gi, '')
   223→   223→}

   456→   456→  async function startResearch(query, researchMode) {
   457→   457→    if (!query.trim() || isResearching) return
   458→   458→
   459→   459→    const id = generateId()
   460→   460→    const research = {
   461→   461→      id,
   462→   462→      query: query.trim(),
   463→   463→      mode: researchMode,
   464→   464→      apexModel,
   465→   465→      sources: [],

   467→   467→      data: null,
   468→   468→      progress: 0,
   469→   469→      stage: 'search',
   470→   470→      timestamp: Date.now(),
   471→   471→    }
   472→   472→

   477→   477→    setSourcesVisible(0)
   478→   478→    setUserScrolledUp(false)
   479→   479→
   480→   480→    // Animate progress
   481→   481→    const stages = researchMode === 'deep' ? DEEP_STAGES : QUICK_STAGES
   482→   482→    let fakeProgress = 0
   483→   483→    const progressInterval = setInterval(() => {
   484→   484→      fakeProgress += Math.random() * 3
   485→   485→      if (fakeProgress > 90) fakeProgress = 90
   486→   486→      setActiveResearch(prev => prev ? { ...prev, progress: Math.min(fakeProgress, prev.stage === 'done' ? 100 : fakeProgress) } : prev)
   487→   487→    }, 500)
   488→   488→

   490→   490→      setTimeout(() => {
   491→   491→        setActiveResearch(prev => prev ? { ...prev, stage: researchMode === 'deep' ? 'analyzing' : 'writing' } : prev)
   492→   492→      }, 2000)
   493→   493→      if (researchMode === 'deep') {
   494→   494→        setTimeout(() => {
   495→   495→          setActiveResearch(prev => prev ? { ...prev, stage: 'writing' } : prev)
   496→   496→        }, 5000)
   497→   497→      }
   498→   498→
   499→   499→      // Call our server-side proxy — it forwards to the Worker with the API key
   500→   500→      // This avoids CORS issues and keeps the API key secure (never exposed to browser)
   501→   501→      const controller = new AbortController()
   502→   502→      const timeout = setTimeout(() => controller.abort(), researchMode === 'deep' ? 180000 : 60000)
   503→   503→
   504→   504→      let res
   505→   505→      let usedFallback = false
   506→   506→      try {
   507→   507→        res = await fetch('/api/research', {
   508→   508→          method: 'POST',
   509→   509→          headers: { 'Content-Type': 'application/json' },
   510→   510→          body: JSON.stringify({ query: query.trim(), mode: researchMode, apex_model: apexModel }),
   511→   511→          signal: controller.signal,
   512→   512→        })
   513→   513→      } finally {
   514→   514→        clearTimeout(timeout)
   515→   515→      }
   516→   516→
   517→   517→      let data = await res.json()
   518→   518→
   519→   519→      // ── Fallback: if worker failed, try Mistral/Groq directly ──
   520→   520→      if (data.error) {
   521→   521→        console.log('[Research] Worker failed, trying fallback providers...', data.error)
   522→   522→        try {
   523→   523→          const fallbackController = new AbortController()
   524→   524→          const fallbackTimeout = setTimeout(() => fallbackController.abort(), 60000)
   525→   525→          const fallbackRes = await fetch('/api/research-fallback', {
   526→   526→            method: 'POST',
   527→   527→            headers: { 'Content-Type': 'application/json' },
   528→   528→            body: JSON.stringify({ query: query.trim(), mode: researchMode, apex_model: apexModel }),
   529→   529→            signal: fallbackController.signal,
   530→   530→          })
   531→   531→          clearTimeout(fallbackTimeout)
   532→   532→          const fallbackData = await fallbackRes.json()
   533→   533→
   534→   534→          if (!fallbackData.error) {
   535→   535→            data = fallbackData
   536→   536→            usedFallback = true
   537→   537→            console.log('[Research] Fallback succeeded')
   538→   538→          } else {
   539→   539→            console.log('[Research] Fallback also failed:', fallbackData.error)
   540→   540→          }
   541→   541→        } catch (fallbackErr) {
   542→   542→          console.log('[Research] Fallback exception:', fallbackErr.message)
   543→   543→        }
   544→   544→      }
   545→   545→
   546→   546→      clearInterval(progressInterval)
   547→   547→
   548→   548→      if (data.error) {
   549→   549→        setError(data.error)
   550→   550→        setIsResearching(false)
   551→   551→        setActiveResearch(prev => prev ? { ...prev, stage: 'done', progress: 100 } : prev)
   552→   552→        return
   553→   553→      }
   554→   554→
   555→   555→      const completed = {
   556→   556→        id,
   557→   557→        query: query.trim(),
   558→   558→        mode: researchMode,
   559→   559→        apexModel,
   560→   560→        sources: data.sources || [],
   561→   561→        report: data.report || '',
   562→   562→        content: data.content || '',
   563→   563→        title: data.title || query.trim(),
   564→   564→        followups: data.followups || [],
   565→   565→        data: data.data || null,
   566→   566→        progress: 100,
   567→   567→        stage: 'done',
   568→   568→        timestamp: Date.now(),
   569→   569→        fallback: usedFallback || data.fallback || false,
   570→   570→        fallbackProvider: data.fallback_provider || null,
   571→   571→        fallbackModel: data.fallback_model || null,
   572→   572→        // APEX 2.0 Wiki fields
   573→   573→        wikiPageId: data.wiki_page_id || data.wikiPageId || null,
   574→   574→        wikiCacheId: data.wiki_cache_id || data.wikiCacheId || null,
   575→   575→        wikiLifecycle: data.wiki_lifecycle || data.wikiLifecycle || 'draft',
   576→   576→        wikiSlug: data.wiki_slug || data.wikiSlug || '',
   577→   577→        wikiVersion: data.wiki_version || data.wikiVersion || 0,
   578→   578→        fromCache: data.from_cache || data.fromCache || false,
   579→   579→      }
   580→   580→
   581→   581→      setActiveResearch(completed)
   582→   582→      setIsResearching(false)
   583→   583→
   584→   584→      if (completed.report) streamReport(completed.report)
   585→   585→      if (completed.sources.length > 0) staggerSources(completed.sources.length)
   586→   586→
   587→   587→      const newHistory = [completed, ...loadHistory()].slice(0, MAX_HISTORY)
   588→   588→      saveHistory(newHistory)
   589→   589→      setHistory(newHistory)
   590→   590→    } catch (err) {
   591→   591→      clearInterval(progressInterval)
   592→   592→      let msg = err.message || 'Research failed. Please try again.'
   593→   593→      if (err.name === 'AbortError') msg = 'Research timed out. Please try again or use Quick mode.'
   594→   594→      else if (msg === 'Failed to fetch') msg = 'Network error. Please check your connection and try again.'
   595→   595→      setError(msg)
   596→   596→      setIsResearching(false)
   597→   597→      setActiveResearch(prev => prev ? { ...prev, stage: 'done', progress: 100 } : prev)
   598→   598→    }
   599→   599→  }
