    90→function groupByDate(sessions) {
    91→  const now = new Date()
    92→  const today = new Date(now.getFullYear(), now.getMonth(), now.getDate())
    93→  const yesterday = new Date(today - 86400000)
    94→  const weekAgo = new Date(today - 7 * 86400000)
    95→
    96→  const groups = { today: [], yesterday: [], week: [], older: [] }
    97→
    98→  sessions.forEach(s => {
    99→    const d = new Date(s.updated_at || s.created_at)
   100→    if (d >= today) groups.today.push(s)
   101→    else if (d >= yesterday) groups.yesterday.push(s)
   102→    else if (d >= weekAgo) groups.week.push(s)
   103→    else groups.older.push(s)
   104→  })
   105→
   106→  const result = []
   107→  if (groups.today.length) result.push({ labelKey: 'chat.date.today', sessions: groups.today })
   108→  if (groups.yesterday.length) result.push({ labelKey: 'chat.date.yesterday', sessions: groups.yesterday })
   109→  if (groups.week.length) result.push({ labelKey: 'chat.date.this_week', sessions: groups.week })
   110→  if (groups.older.length) result.push({ labelKey: 'chat.date.older', sessions: groups.older })
   111→  return result
   112→}

   450→  // ── Feature #6: File Upload Handler ──
   451→  function handleFileSelect(e) {
   452→    const file = e.target.files?.[0]
   453→    if (!file) return
   454→
   455→    const ext = '.' + file.name.split('.').pop().toLowerCase()
   456→    const isImage = ALLOWED_IMAGE_EXTENSIONS.includes(ext)
   457→    const isText = ALLOWED_TEXT_EXTENSIONS.includes(ext)
   458→
   459→    if (!isImage && !isText) {
   460→      alert(t('chat.error.unsupported_file'))
   461→      e.target.value = ''
   462→      return
   463→    }
   464→
   465→    if (file.size > MAX_FILE_SIZE) {
   466→      alert(t('chat.error.file_too_large'))
   467→      e.target.value = ''
   468→      return
   469→    }
   470→
   471→    if (isImage) {
   472→      // Read as base64 for vision model
   473→      const reader = new FileReader()
   474→      reader.onload = () => {
   475→        setAttachedFile(file.name)
   476→        setAttachedContent(reader.result) // data:image/...;base64,...
   477→        setAttachedIsImage(true)
   478→      }
   479→      reader.readAsDataURL(file)
   480→    } else {
   481→      // Read as text
   482→      const reader = new FileReader()
   483→      reader.onload = () => {
   484→        setAttachedFile(file.name)
   485→        setAttachedContent(reader.result)
   486→        setAttachedIsImage(false)
   487→      }
   488→      reader.readAsText(file)
   489→    }
   490→    e.target.value = ''
   491→  }

   501→    if (messages.length === 0) return
   502→    const now = new Date()
   503→    const dateStr = now.toISOString().split('T')[0]
   504→    const timeStr = now.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })
   505→    const currentModel = MODELS.find(m => m.id === model)
   506→    const modelName = currentModel?.name || model
   507→
   508→    try {
   509→      const { exportChatAsPDF, exportChatAsDOCX, downloadBlob } = await import('@/lib/fileExportHeavy')
   510→      const { exportChatAsHTML } = await import('@/lib/fileExportClient')
   511→
   512→      if (format === 'pdf') {
   513→        const blob = await exportChatAsPDF(messages, modelName)
   514→        downloadBlob(blob, `kivora-chat-${dateStr}.pdf`)
   515→      } else if (format === 'docx') {
   516→        const blob = await exportChatAsDOCX(messages, modelName)
   517→        downloadBlob(blob, `kivora-chat-${dateStr}.docx`)
   518→      } else if (format === 'html') {
   519→        const blob = exportChatAsHTML(messages, modelName)
   520→        downloadBlob(blob, `kivora-chat-${dateStr}.html`)
   521→      } else {
   522→        // MD and TXT — no external deps needed
   523→        let content = ''
   524→        if (format === 'md') {
   525→          content = `# Kivora AI Chat Export\n\n`
   526→          content += `**Date:** ${dateStr} ${timeStr}\n`
   527→          content += `**Model:** ${modelName}\n\n`
   528→          content += `---\n\n`
   529→          messages.forEach(msg => {
   530→            if (msg.role === 'user') {
   531→              content += `### User\n\n${msg.content}\n\n`
   532→            } else {
   533→              content += `### Assistant\n\n${msg.content}\n\n`
   534→            }
   535→          })
   536→        } else {
   537→          content = `Kivora AI Chat Export\n`
   538→          content += `Date: ${dateStr} ${timeStr}\n`
   539→          content += `Model: ${modelName}\n`
   540→          content += `${'='.repeat(50)}\n\n`
   541→          messages.forEach(msg => {
   542→            if (msg.role === 'user') {
   543→              content += `[User]\n${msg.content}\n\n`
   544→            } else {
   545→              content += `[Assistant]\n${msg.content}\n\n`
   546→            }
   547→          })
   548→        }
   549→        const blob = new Blob([content], { type: format === 'md' ? 'text/markdown' : 'text/plain' })
   550→        downloadBlob(blob, `kivora-chat-${dateStr}.${format}`)
   551→      }
   552→    } catch (err) {
   553→      console.error('Export failed:', err)
   554→    }
   555→  }

   697→  async function checkMicPermission() {
   698→    try {
   699→      const result = await navigator.permissions.query({ name: 'microphone' })
   700→      return result.state // 'granted', 'denied', or 'prompt'
   701→    } catch {
   702→      // permissions.query not supported in some browsers — return null so we fall through to getUserMedia
   703→      return null
   704→    }

   707→  async function toggleListening(e) {
   708→    // Do NOT call e.preventDefault() — it blocks the browser's permission popup
   709→    // by cancelling the "user gesture" that the browser requires.
   710→    if (e) { e.stopPropagation() }
   711→    if (typeof window === 'undefined') return
   712→    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition
   713→
   714→    if (!SpeechRecognition) {
   715→      showVoiceToast('error', t('chat.voice_not_supported') || 'Voice input is not supported in this browser. Please use Chrome or Edge.')
   716→      return
   717→    }
   718→
   719→    if (isListening) {
   720→      if (recognitionRef.current) {
   721→        recognitionRef.current.stop()
   722→      }
   723→      setIsListening(false)
   724→      return
   725→    }
   726→
   727→    // Step 1: Check if permission is already permanently denied.
   728→    // If so, getUserMedia will silently reject without showing a popup,
   729→    // so we must show a modal with manual reset instructions instead.
   730→    const permState = await checkMicPermission()
   731→    if (permState === 'denied') {
   732→      setMicPermModal(true)
   733→      return
   734→    }
   735→
   736→    // Step 2: If permission is 'prompt' (never asked) or 'granted', proceed.
   737→    // For 'prompt': getUserMedia will trigger the browser's permission popup.
   738→    // For 'granted': getUserMedia will succeed immediately.
   739→    // If permissions.query is unsupported (permState === null), try getUserMedia anyway.
   740→    if (permState === 'prompt' || permState === null) {
   741→      try {
   742→        const stream = await navigator.mediaDevices.getUserMedia({ audio: true })
   743→        // Release the stream immediately — we only needed the permission grant
   744→        stream.getTracks().forEach(track => track.stop())
   745→      } catch (err) {
   746→        if (err.name === 'NotAllowedError' || err.name === 'PermissionDeniedError') {
   747→          // User just clicked "Block" on the popup, or it was auto-denied
   748→          setMicPermModal(true)
   749→        } else if (err.name === 'NotFoundError') {
   750→          showVoiceToast('error', t('chat.voice_no_mic') || 'No microphone found. Please connect a microphone.')
   751→        } else {
   752→          showVoiceToast('error', t('chat.voice_error') || 'Voice input error. Please try again.')
   753→        }
   754→        return
   755→      }
   756→    }
   757→
   758→    // Step 3: Permission is granted — start SpeechRecognition
   759→    const recognition = new SpeechRecognition()
   760→    recognition.continuous = true
   761→    recognition.interimResults = true
   762→    recognition.lang = 'en-US'
   763→
   764→    let finalTranscript = ''
   765→
   766→    recognition.onresult = (event) => {
   767→      for (let i = event.resultIndex; i < event.results.length; i++) {
   768→        const transcript = event.results[i][0].transcript
   769→        if (event.results[i].isFinal) {
   770→          finalTranscript += transcript
   771→        }
   772→      }
   773→      if (finalTranscript) {
   774→        setInput(prev => prev ? prev + ' ' + finalTranscript : finalTranscript)
   775→        finalTranscript = ''
   776→        if (textareaRef.current) {
   777→          setTimeout(() => autoResize(textareaRef.current), 0)
   778→        }
   779→      }
   780→    }
   781→
   782→    recognition.onerror = (event) => {
   783→      setIsListening(false)
   784→      if (event.error === 'not-allowed') {
   785→        setMicPermModal(true)
   786→      } else if (event.error === 'no-speech') {
   787→        showVoiceToast('info', t('chat.voice_no_speech') || 'No speech detected. Please try again.')
   788→      } else if (event.error === 'audio-capture') {
   789→        showVoiceToast('error', t('chat.voice_no_mic') || 'No microphone found. Please connect a microphone.')
   790→      } else {
   791→        showVoiceToast('error', t('chat.voice_error') || 'Voice input error. Please try again.')
   792→      }
   793→    }
   794→
   795→    recognition.onend = () => {
   796→      setIsListening(false)
   797→    }
   798→
   799→    recognitionRef.current = recognition
   800→    setIsListening(true)
   801→    try {
   802→      recognition.start()
   803→      showVoiceToast('info', t('chat.voice_listening') || 'Listening... Speak now.')
   804→    } catch (e) {
   805→      setIsListening(false)
   806→      showVoiceToast('error', t('chat.voice_error') || 'Voice input error. Please try again.')
   807→    }
   808→  }
