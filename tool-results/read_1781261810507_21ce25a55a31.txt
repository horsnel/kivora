    90→function groupByDate(sessions) {
    91→  const now = new Date()
    92→  const today = new Date(now.getFullYear(), now.getMonth(), now.getDate())
    93→  const yesterday = new Date(today - 86400000)
    94→  const weekAgo = new Date(today - 7 * 86400000)
    95→
    96→  const groups = { today: [], yesterday: [], week: [], older: [] }
    97→
    98→  sessions.forEach(s => {
    99→    const d = new Date(s.updated_at || s.created_at)
   100→    if (d >= today) groups.today.push(s)
   101→    else if (d >= yesterday) groups.yesterday.push(s)
   102→    else if (d >= weekAgo) groups.week.push(s)
   103→    else groups.older.push(s)
   104→  })
   105→
   106→  const result = []
   107→  if (groups.today.length) result.push({ labelKey: 'chat.date.today', sessions: groups.today })
   108→  if (groups.yesterday.length) result.push({ labelKey: 'chat.date.yesterday', sessions: groups.yesterday })
   109→  if (groups.week.length) result.push({ labelKey: 'chat.date.this_week', sessions: groups.week })
   110→  if (groups.older.length) result.push({ labelKey: 'chat.date.older', sessions: groups.older })
   111→  return result
   112→}

   464→  // ── Feature #6: File Upload Handler ──
   465→  function handleFileSelect(e) {
   466→    const file = e.target.files?.[0]
   467→    if (!file) return
   468→
   469→    const ext = '.' + file.name.split('.').pop().toLowerCase()
   470→    const isImage = ALLOWED_IMAGE_EXTENSIONS.includes(ext)
   471→    const isText = ALLOWED_TEXT_EXTENSIONS.includes(ext)
   472→
   473→    if (!isImage && !isText) {
   474→      alert(t('chat.error.unsupported_file'))
   475→      e.target.value = ''
   476→      return
   477→    }
   478→
   479→    if (file.size > MAX_FILE_SIZE) {
   480→      alert(t('chat.error.file_too_large'))
   481→      e.target.value = ''
   482→      return
   483→    }
   484→
   485→    if (isImage) {
   486→      // Read as base64 for vision model
   487→      const reader = new FileReader()
   488→      reader.onload = () => {
   489→        setAttachedFile(file.name)
   490→        setAttachedContent(reader.result) // data:image/...;base64,...
   491→        setAttachedIsImage(true)
   492→      }
   493→      reader.readAsDataURL(file)
   494→    } else {
   495→      // Read as text

   514→  async function exportChat(format) {
   515→    if (messages.length === 0) return
   516→    const now = new Date()
   517→    const dateStr = now.toISOString().split('T')[0]
   518→    const timeStr = now.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })
   519→    const currentModel = MODELS.find(m => m.id === model)
   520→    const modelName = currentModel?.name || model
   521→
   522→    try {
   523→      const { exportChatAsPDF, exportChatAsDOCX, downloadBlob } = await import('@/lib/fileExportHeavy')
   524→      const { exportChatAsHTML } = await import('@/lib/fileExportClient')
   525→
   526→      if (format === 'pdf') {
   527→        const blob = await exportChatAsPDF(messages, modelName)
   528→        downloadBlob(blob, `kivora-chat-${dateStr}.pdf`)
   529→      } else if (format === 'docx') {
   530→        const blob = await exportChatAsDOCX(messages, modelName)
   531→        downloadBlob(blob, `kivora-chat-${dateStr}.docx`)
   532→      } else if (format === 'html') {
   533→        const blob = exportChatAsHTML(messages, modelName)
   534→        downloadBlob(blob, `kivora-chat-${dateStr}.html`)
   535→      } else {
   536→        // MD and TXT — no external deps needed
   537→        let content = ''
   538→        if (format === 'md') {
   539→          content = `# Kivora AI Chat Export\n\n`
   540→          content += `**Date:** ${dateStr} ${timeStr}\n`
   541→          content += `**Model:** ${modelName}\n\n`
   542→          content += `---\n\n`
   543→          messages.forEach(msg => {
   544→            if (msg.role === 'user') {
   545→              content += `### User\n\n${msg.content}\n\n`
   546→            } else {
   547→              content += `### Assistant\n\n${msg.content}\n\n`
   548→            }
   549→          })
   550→        } else {
   551→          content = `Kivora AI Chat Export\n`
   552→          content += `Date: ${dateStr} ${timeStr}\n`
   553→          content += `Model: ${modelName}\n`
   554→          content += `${'='.repeat(50)}\n\n`
   555→          messages.forEach(msg => {
   556→            if (msg.role === 'user') {
   557→              content += `[User]\n${msg.content}\n\n`
   558→            } else {
   559→              content += `[Assistant]\n${msg.content}\n\n`
   560→            }
   561→          })
   562→        }
   563→        const blob = new Blob([content], { type: format === 'md' ? 'text/markdown' : 'text/plain' })
   564→        downloadBlob(blob, `kivora-chat-${dateStr}.${format}`)
   565→      }
   566→    } catch (err) {
   567→      console.error('Export failed:', err)
   568→    }

   711→  async function checkMicPermission() {
   712→    try {
   713→      const result = await navigator.permissions.query({ name: 'microphone' })
   714→      return result.state // 'granted', 'denied', or 'prompt'

   721→  async function toggleListening(e) {
   722→    // Do NOT call e.preventDefault() — it blocks the browser's permission popup
   723→    // by cancelling the "user gesture" that the browser requires.
   724→    if (e) { e.stopPropagation() }
   725→    if (typeof window === 'undefined') return
   726→    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition
   727→
   728→    if (!SpeechRecognition) {
   729→      showVoiceToast('error', t('chat.voice_not_supported') || 'Voice input is not supported in this browser. Please use Chrome or Edge.')
   730→      return
   731→    }
   732→
   733→    if (isListening) {
   734→      if (recognitionRef.current) {
   735→        recognitionRef.current.stop()
   736→      }
   737→      setIsListening(false)
   738→      return
   739→    }
   740→
   741→    // Step 1: Check if permission is already permanently denied.
   742→    // If so, getUserMedia will silently reject without showing a popup,
   743→    // so we must show a modal with manual reset instructions instead.
   744→    const permState = await checkMicPermission()
   745→    if (permState === 'denied') {
   746→      setMicPermModal(true)
   747→      return
   748→    }
   749→
   750→    // Step 2: If permission is 'prompt' (never asked) or 'granted', proceed.
   751→    // For 'prompt': getUserMedia will trigger the browser's permission popup.
   752→    // For 'granted': getUserMedia will succeed immediately.
   753→    // If permissions.query is unsupported (permState === null), try getUserMedia anyway.
   754→    if (permState === 'prompt' || permState === null) {
   755→      try {
   756→        const stream = await navigator.mediaDevices.getUserMedia({ audio: true })
   757→        // Release the stream immediately — we only needed the permission grant
   758→        stream.getTracks().forEach(track => track.stop())
   759→      } catch (err) {
   760→        if (err.name === 'NotAllowedError' || err.name === 'PermissionDeniedError') {
   761→          // User just clicked "Block" on the popup, or it was auto-denied
   762→          setMicPermModal(true)
   763→        } else if (err.name === 'NotFoundError') {
   764→          showVoiceToast('error', t('chat.voice_no_mic') || 'No microphone found. Please connect a microphone.')
   765→        } else {
   766→          showVoiceToast('error', t('chat.voice_error') || 'Voice input error. Please try again.')
   767→        }
   768→        return
   769→      }
   770→    }
   771→
   772→    // Step 3: Permission is granted — start SpeechRecognition
   773→    const recognition = new SpeechRecognition()
   774→    recognition.continuous = true
   775→    recognition.interimResults = true
   776→    recognition.lang = 'en-US'
   777→
   778→    let finalTranscript = ''
   779→
   780→    recognition.onresult = (event) => {
   781→      for (let i = event.resultIndex; i < event.results.length; i++) {
   782→        const transcript = event.results[i][0].transcript
   783→        if (event.results[i].isFinal) {
   784→          finalTranscript += transcript
   785→        }
   786→      }
   787→      if (finalTranscript) {
   788→        setInput(prev => prev ? prev + ' ' + finalTranscript : finalTranscript)
   789→        finalTranscript = ''
   790→        if (textareaRef.current) {
   791→          setTimeout(() => autoResize(textareaRef.current), 0)
   792→        }
   793→      }
   794→    }
   795→
   796→    recognition.onerror = (event) => {
   797→      setIsListening(false)
   798→      if (event.error === 'not-allowed') {
   799→        setMicPermModal(true)
   800→      } else if (event.error === 'no-speech') {
   801→        showVoiceToast('info', t('chat.voice_no_speech') || 'No speech detected. Please try again.')
   802→      } else if (event.error === 'audio-capture') {
   803→        showVoiceToast('error', t('chat.voice_no_mic') || 'No microphone found. Please connect a microphone.')
   804→      } else {
   805→        showVoiceToast('error', t('chat.voice_error') || 'Voice input error. Please try again.')
   806→      }
   807→    }
   808→
   809→    recognition.onend = () => {
   810→      setIsListening(false)
   811→    }
   812→
   813→    recognitionRef.current = recognition
   814→    setIsListening(true)
   815→    try {
   816→      recognition.start()
   817→      showVoiceToast('info', t('chat.voice_listening') || 'Listening... Speak now.')
   818→    } catch (e) {
   819→      setIsListening(false)
   820→      showVoiceToast('error', t('chat.voice_error') || 'Voice input error. Please try again.')
   821→    }
   822→  }
