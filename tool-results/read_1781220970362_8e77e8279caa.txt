    89→function groupByDate(sessions) {
    90→  const now = new Date()
    91→  const today = new Date(now.getFullYear(), now.getMonth(), now.getDate())
    92→  const yesterday = new Date(today - 86400000)
    93→  const weekAgo = new Date(today - 7 * 86400000)
    94→
    95→  const groups = { today: [], yesterday: [], week: [], older: [] }
    96→
    97→  sessions.forEach(s => {
    98→    const d = new Date(s.updated_at || s.created_at)
    99→    if (d >= today) groups.today.push(s)
   100→    else if (d >= yesterday) groups.yesterday.push(s)
   101→    else if (d >= weekAgo) groups.week.push(s)
   102→    else groups.older.push(s)
   103→  })
   104→
   105→  const result = []
   106→  if (groups.today.length) result.push({ labelKey: 'chat.date.today', sessions: groups.today })
   107→  if (groups.yesterday.length) result.push({ labelKey: 'chat.date.yesterday', sessions: groups.yesterday })
   108→  if (groups.week.length) result.push({ labelKey: 'chat.date.this_week', sessions: groups.week })
   109→  if (groups.older.length) result.push({ labelKey: 'chat.date.older', sessions: groups.older })
   110→  return result
   111→}

   441→  function handleFileSelect(e) {
   442→    const file = e.target.files?.[0]
   443→    if (!file) return
   444→
   445→    const ext = '.' + file.name.split('.').pop().toLowerCase()
   446→    const isImage = ALLOWED_IMAGE_EXTENSIONS.includes(ext)
   447→    const isText = ALLOWED_TEXT_EXTENSIONS.includes(ext)
   448→
   449→    if (!isImage && !isText) {
   450→      alert(t('chat.error.unsupported_file'))
   451→      e.target.value = ''
   452→      return
   453→    }
   454→
   455→    if (file.size > MAX_FILE_SIZE) {
   456→      alert(t('chat.error.file_too_large'))
   457→      e.target.value = ''
   458→      return
   459→    }
   460→
   461→    if (isImage) {
   462→      // Read as base64 for vision model
   463→      const reader = new FileReader()
   464→      reader.onload = () => {
   465→        setAttachedFile(file.name)
   466→        setAttachedContent(reader.result) // data:image/...;base64,...
   467→        setAttachedIsImage(true)
   468→      }
   469→      reader.readAsDataURL(file)
   470→    } else {
   471→      // Read as text
   472→      const reader = new FileReader()
   473→      reader.onload = () => {
   474→        setAttachedFile(file.name)
   475→        setAttachedContent(reader.result)
   476→        setAttachedIsImage(false)
   477→      }
   478→      reader.readAsText(file)
   479→    }
   480→    e.target.value = ''
   481→  }

   687→  async function checkMicPermission() {
   688→    try {
   689→      const result = await navigator.permissions.query({ name: 'microphone' })
   690→      return result.state // 'granted', 'denied', or 'prompt'
   691→    } catch {

   693→      return null
   694→    }

   697→  async function toggleListening(e) {
   698→    // Do NOT call e.preventDefault() — it blocks the browser's permission popup
   699→    // by cancelling the "user gesture" that the browser requires.
   700→    if (e) { e.stopPropagation() }
   701→    if (typeof window === 'undefined') return
   702→    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition
   703→
   704→    if (!SpeechRecognition) {
   705→      showVoiceToast('error', t('chat.voice_not_supported') || 'Voice input is not supported in this browser. Please use Chrome or Edge.')
   706→      return
   707→    }
   708→
   709→    if (isListening) {
   710→      if (recognitionRef.current) {
   711→        recognitionRef.current.stop()
   712→      }
   713→      setIsListening(false)
   714→      return
   715→    }
   716→
   717→    // Step 1: Check if permission is already permanently denied.
   718→    // If so, getUserMedia will silently reject without showing a popup,
   719→    // so we must show a modal with manual reset instructions instead.
   720→    const permState = await checkMicPermission()
   721→    if (permState === 'denied') {
   722→      setMicPermModal(true)
   723→      return
   724→    }
   725→
   726→    // Step 2: If permission is 'prompt' (never asked) or 'granted', proceed.
   727→    // For 'prompt': getUserMedia will trigger the browser's permission popup.
   728→    // For 'granted': getUserMedia will succeed immediately.
   729→    // If permissions.query is unsupported (permState === null), try getUserMedia anyway.
   730→    if (permState === 'prompt' || permState === null) {
   731→      try {
   732→        const stream = await navigator.mediaDevices.getUserMedia({ audio: true })
   733→        // Release the stream immediately — we only needed the permission grant
   734→        stream.getTracks().forEach(track => track.stop())
   735→      } catch (err) {
   736→        if (err.name === 'NotAllowedError' || err.name === 'PermissionDeniedError') {
   737→          // User just clicked "Block" on the popup, or it was auto-denied
   738→          setMicPermModal(true)
   739→        } else if (err.name === 'NotFoundError') {
   740→          showVoiceToast('error', t('chat.voice_no_mic') || 'No microphone found. Please connect a microphone.')
   741→        } else {
   742→          showVoiceToast('error', t('chat.voice_error') || 'Voice input error. Please try again.')
   743→        }
   744→        return
   745→      }
   746→    }
   747→
   748→    // Step 3: Permission is granted — start SpeechRecognition
   749→    const recognition = new SpeechRecognition()
   750→    recognition.continuous = true
   751→    recognition.interimResults = true
   752→    recognition.lang = 'en-US'
   753→
   754→    let finalTranscript = ''
   755→
   756→    recognition.onresult = (event) => {
   757→      for (let i = event.resultIndex; i < event.results.length; i++) {
   758→        const transcript = event.results[i][0].transcript
   759→        if (event.results[i].isFinal) {
   760→          finalTranscript += transcript
   761→        }
   762→      }
   763→      if (finalTranscript) {
   764→        setInput(prev => prev ? prev + ' ' + finalTranscript : finalTranscript)
   765→        finalTranscript = ''
   766→        if (textareaRef.current) {
   767→          setTimeout(() => autoResize(textareaRef.current), 0)
   768→        }
   769→      }
   770→    }
   771→
   772→    recognition.onerror = (event) => {
   773→      setIsListening(false)
   774→      if (event.error === 'not-allowed') {
   775→        setMicPermModal(true)
   776→      } else if (event.error === 'no-speech') {
   777→        showVoiceToast('info', t('chat.voice_no_speech') || 'No speech detected. Please try again.')
   778→      } else if (event.error === 'audio-capture') {
   779→        showVoiceToast('error', t('chat.voice_no_mic') || 'No microphone found. Please connect a microphone.')
   780→      } else {
   781→        showVoiceToast('error', t('chat.voice_error') || 'Voice input error. Please try again.')
   782→      }
   783→    }
   784→
   785→    recognition.onend = () => {
   786→      setIsListening(false)
   787→    }
   788→
   789→    recognitionRef.current = recognition
   790→    setIsListening(true)
   791→    try {
   792→      recognition.start()
   793→      showVoiceToast('info', t('chat.voice_listening') || 'Listening... Speak now.')
   794→    } catch (e) {
   795→      setIsListening(false)
   796→      showVoiceToast('error', t('chat.voice_error') || 'Voice input error. Please try again.')
   797→    }
   798→  }
