    72→// ── Heading patterns that indicate a duplicate Sources/References section ──
    73→const DUPLICATE_SOURCE_HEADINGS = /^(#\s*)?(sources|references|bibliography|citations|works?\s+cited)\s*$/i

    75→// ── Simple Markdown Renderer ──
    76→function renderMarkdown(text, { skipDuplicateSources = false } = {}) {
    77→  if (!text) return null
    78→  const lines = text.split('\n')
    79→  const elements = []
    80→  let inList = false
    81→  let listItems = []
    82→  let inTable = false
    83→  let tableRows = []
    84→  let tableHeaders = []
    85→  let keyIdx = 0
    86→  let skipMode = false // When true, we're inside a duplicate Sources section — skip all content
    87→
    88→  function flushList() {
    89→    if (listItems.length > 0) {
    90→      elements.push(<ul key={`ul-${keyIdx++}`} className="space-y-1 mb-3 ml-1">{listItems}</ul>)
    91→      listItems = []
    92→    }
    93→    inList = false
    94→  }
    95→
    96→  function flushTable() {
    97→    if (tableHeaders.length > 0) {
    98→      elements.push(
    99→        <div key={`tbl-wrap-${keyIdx++}`} className="-mx-1 px-1 mb-3">
   100→          <table className="w-full border-collapse">
   101→            <thead>
   102→              <tr>{tableHeaders.map((h, hi) => (
   103→                <th key={`th-${hi}`}>{h}</th>
   104→              ))}</tr>
   105→            </thead>
   106→            <tbody>
   107→              {tableRows.map((row, ri) => (
   108→                <tr key={`tr-${ri}`}>{row.map((cell, ci) => (
   109→                  <td key={`td-${ri}-${ci}`}>{renderInline(cell)}</td>
   110→                ))}</tr>
   111→              ))}
   112→            </tbody>
   113→          </table>
   114→        </div>
   115→      )
   116→    }
   117→    tableHeaders = []
   118→    tableRows = []
   119→    inTable = false
   120→  }
   121→
   122→  function renderInline(str) {
   123→    const parts = []
   124→    const regex = /(\*\*|__)(.*?)\1|\[([^\]]+)\]\(([^)]+)\)|\[(\d+)\]/g
   125→    let lastIndex = 0
   126→    let match
   127→    while ((match = regex.exec(str)) !== null) {
   128→      if (match.index > lastIndex) {
   129→        parts.push(str.slice(lastIndex, match.index))
   130→      }
   131→      if (match[1]) {
   132→        parts.push(<strong key={`b-${keyIdx++}`} className="text-white font-semibold">{match[2]}</strong>)
   133→      } else if (match[3] !== undefined && match[4] !== undefined) {
   134→        parts.push(
   135→          <a key={`a-${keyIdx++}`} href={match[4]} target="_blank" rel="noopener noreferrer" className="text-red-400 hover:text-red-300 underline underline-offset-2">
   136→            {match[3]}
   137→          </a>
   138→        )
   139→      } else if (match[5] !== undefined) {
   140→        parts.push(<span key={`cit-${keyIdx++}`} className="citation-number">[{match[5]}]</span>)
   141→      }
   142→      lastIndex = match.index + match[0].length
   143→    }
   144→    if (lastIndex < str.length) parts.push(str.slice(lastIndex))
   145→    return parts.length > 0 ? parts : str
   146→  }
   147→
   148→  for (let i = 0; i < lines.length; i++) {
   149→    const line = lines[i]
   150→
   151→    // Detect duplicate Sources/References heading — skip everything until the next heading of same or higher level
   152→    if (skipDuplicateSources && DUPLICATE_SOURCE_HEADINGS.test(line.trim())) {
   153→      flushList()
   154→      flushTable()
   155→      const headingLevel = line.match(/^(#{1,3})\s/)?.[1]?.length || 0
   156→      skipMode = headingLevel
   157→      continue
   158→    }
   159→    // Exit skip mode when we hit a new heading at same or higher level
   160→    if (skipMode) {
   161→      const newHeadingMatch = line.match(/^(#{1,3})\s/)
   162→      if (newHeadingMatch) {
   163→        const newLevel = newHeadingMatch[1].length
   164→        if (newLevel <= skipMode) {
   165→          skipMode = false
   166→          // Don't continue — process this heading normally
   167→        } else {
   168→          continue // Still inside the duplicate section (sub-heading)
   169→        }
   170→      } else if (/^---+$/.test(line.trim())) {
   171→        // Horizontal rule might end the section
   172→        skipMode = false
   173→        continue
   174→      } else {
   175→        continue // Skip content inside duplicate section
   176→      }
   177→    }
   178→
   179→    // Table row detection: lines containing pipes
   180→    if (line.includes('|') && line.trim().startsWith('|')) {
   181→      flushList()
   182→      const cells = line.split('|').filter(c => c.trim() !== '').map(c => c.trim())
   183→      // Separator row (|---|---|)
   184→      if (cells.every(c => /^[-:]+$/.test(c))) {
   185→        inTable = true
   186→        continue
   187→      }
   188→      if (!inTable && tableHeaders.length === 0) {
   189→        // First row = header
   190→        tableHeaders = cells
   191→        inTable = true
   192→        continue
   193→      }
   194→      if (inTable) {
   195→        tableRows.push(cells)
   196→        continue
   197→      }
   198→    } else if (inTable) {
   199→      // End of table
   200→      flushTable()
   201→    }
   202→
   203→    if (line.startsWith('### ')) { flushList(); elements.push(<h3 key={`h3-${keyIdx++}`} className="text-base font-semibold text-white mt-5 mb-2">{renderInline(line.slice(4))}</h3>); continue }
   204→    if (line.startsWith('## ')) { flushList(); elements.push(<h2 key={`h2-${keyIdx++}`} className="text-lg font-semibold text-white mt-6 mb-2.5">{renderInline(line.slice(3))}</h2>); continue }
   205→    if (line.startsWith('# ')) { flushList(); elements.push(<h1 key={`h1-${keyIdx++}`} className="text-xl font-bold text-white mt-4 mb-3">{renderInline(line.slice(2))}</h1>); continue }
   206→    if (/^---+$/.test(line.trim())) { flushList(); elements.push(<hr key={`hr-${keyIdx++}`} className="border-[#1a1a1a] my-4" />); continue }
   207→    if (/^[-*]\s/.test(line)) { inList = true; listItems.push(<li key={`li-${keyIdx++}`} className="text-sm text-[#a0a0a0] leading-relaxed pl-1">{renderInline(line.replace(/^[-*]\s/, ''))}</li>); continue }
   208→    if (/^\d+\.\s/.test(line)) { inList = true; listItems.push(<li key={`li-${keyIdx++}`} className="text-sm text-[#a0a0a0] leading-relaxed pl-1 list-decimal ml-4">{renderInline(line.replace(/^\d+\.\s/, ''))}</li>); continue }
   209→    if (!line.trim()) { flushList(); continue }
   210→    flushList()
   211→    elements.push(<p key={`p-${keyIdx++}`} className="text-sm text-[#a0a0a0] leading-relaxed mb-3">{renderInline(line)}</p>)
   212→  }
   213→  flushList()
   214→  flushTable()
   215→  return elements
   216→}

   218→// ── Filter duplicate Sources/References from HTML content (dangerouslySetInnerHTML path) ──
   219→function filterDuplicateSources(html) {
   220→  if (!html) return html
   221→  // Remove sections headed by Sources/References/Bibliography etc.
   222→  return html.replace(/<h[1-3][^>]*>\s*(Sources|References|Bibliography|Citations|Works?\s+Cited)\s*<\/h[1-3]>[\s\S]*?(?=<h[1-3][^>]*>|$)/gi, '')
   223→}

   513→      } finally {

   515→      }
   516→
   517→      let data = await res.json()
   518→
   519→      // ── Fallback: if worker failed, try Mistral/Groq directly ──
   520→      if (data.error) {
   521→        console.log('[Research] Worker failed, trying fallback providers...', data.error)
   522→        try {
   523→          const fallbackController = new AbortController()

   526→            method: 'POST',
   527→            headers: { 'Content-Type': 'application/json' },
   528→            body: JSON.stringify({ query: query.trim(), mode: researchMode, apex_model: apexModel }),

   534→          if (!fallbackData.error) {
   535→            data = fallbackData
   536→            usedFallback = true
   537→            console.log('[Research] Fallback succeeded')
   538→          } else {
   539→            console.log('[Research] Fallback also failed:', fallbackData.error)
   540→          }
   541→        } catch (fallbackErr) {
   542→          console.log('[Research] Fallback exception:', fallbackErr.message)
   543→        }
   544→      }
   545→
   546→      clearInterval(progressInterval)
   547→
   548→      if (data.error) {
   549→        setError(data.error)
   550→        setIsResearching(false)
   551→        setActiveResearch(prev => prev ? { ...prev, stage: 'done', progress: 100 } : prev)
   552→        return
   553→      }
   554→
   555→      const completed = {
   556→        id,
   557→        query: query.trim(),
   558→        mode: researchMode,
   559→        apexModel,
   560→        sources: data.sources || [],
   561→        report: data.report || '',
   562→        content: data.content || '',
   563→        title: data.title || query.trim(),
   564→        followups: data.followups || [],
   565→        data: data.data || null,
   566→        progress: 100,
   567→        stage: 'done',
   568→        timestamp: Date.now(),
   569→        fallback: usedFallback || data.fallback || false,
   570→        fallbackProvider: data.fallback_provider || null,
   571→        fallbackModel: data.fallback_model || null,
   572→        // APEX 2.0 Wiki fields
   573→        wikiPageId: data.wiki_page_id || data.wikiPageId || null,
   574→        wikiCacheId: data.wiki_cache_id || data.wikiCacheId || null,

   577→        wikiVersion: data.wiki_version || data.wikiVersion || 0,
   578→        fromCache: data.from_cache || data.fromCache || false,
   579→      }

   582→      setIsResearching(false)

   586→
   587→      const newHistory = [completed, ...loadHistory()].slice(0, MAX_HISTORY)
   588→      saveHistory(newHistory)
   589→      setHistory(newHistory)
   590→    } catch (err) {
   591→      clearInterval(progressInterval)
   592→      let msg = err.message || 'Research failed. Please try again.'
   593→      if (err.name === 'AbortError') msg = 'Research timed out. Please try again or use Quick mode.'
   594→      else if (msg === 'Failed to fetch') msg = 'Network error. Please check your connection and try again.'
   595→      setError(msg)
   596→      setIsResearching(false)
   597→      setActiveResearch(prev => prev ? { ...prev, stage: 'done', progress: 100 } : prev)
   598→    }
   599→  }
