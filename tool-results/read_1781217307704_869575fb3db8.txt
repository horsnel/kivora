    89→function groupByDate(sessions) {
    90→  const now = new Date()
    91→  const today = new Date(now.getFullYear(), now.getMonth(), now.getDate())
    92→  const yesterday = new Date(today - 86400000)
    93→  const weekAgo = new Date(today - 7 * 86400000)
    94→
    95→  const groups = { today: [], yesterday: [], week: [], older: [] }
    96→
    97→  sessions.forEach(s => {
    98→    const d = new Date(s.updated_at || s.created_at)
    99→    if (d >= today) groups.today.push(s)
   100→    else if (d >= yesterday) groups.yesterday.push(s)
   101→    else if (d >= weekAgo) groups.week.push(s)
   102→    else groups.older.push(s)
   103→  })
   104→
   105→  const result = []
   106→  if (groups.today.length) result.push({ labelKey: 'chat.date.today', sessions: groups.today })
   107→  if (groups.yesterday.length) result.push({ labelKey: 'chat.date.yesterday', sessions: groups.yesterday })
   108→  if (groups.week.length) result.push({ labelKey: 'chat.date.this_week', sessions: groups.week })
   109→  if (groups.older.length) result.push({ labelKey: 'chat.date.older', sessions: groups.older })
   110→  return result
   111→}

   414→  // ── Feature #6: File Upload Handler ──
   415→  function handleFileSelect(e) {
   416→    const file = e.target.files?.[0]
   417→    if (!file) return
   418→
   419→    const ext = '.' + file.name.split('.').pop().toLowerCase()
   420→    const isImage = ALLOWED_IMAGE_EXTENSIONS.includes(ext)
   421→    const isText = ALLOWED_TEXT_EXTENSIONS.includes(ext)
   422→
   423→    if (!isImage && !isText) {
   424→      alert(t('chat.error.unsupported_file'))
   425→      e.target.value = ''
   426→      return
   427→    }
   428→
   429→    if (file.size > MAX_FILE_SIZE) {
   430→      alert(t('chat.error.file_too_large'))
   431→      e.target.value = ''
   432→      return
   433→    }
   434→
   435→    if (isImage) {
   436→      // Read as base64 for vision model
   437→      const reader = new FileReader()
   438→      reader.onload = () => {
   439→        setAttachedFile(file.name)
   440→        setAttachedContent(reader.result) // data:image/...;base64,...
   441→        setAttachedIsImage(true)
   442→      }
   443→      reader.readAsDataURL(file)
   444→    } else {
   445→      // Read as text
   446→      const reader = new FileReader()
   447→      reader.onload = () => {
   448→        setAttachedFile(file.name)
   449→        setAttachedContent(reader.result)
   450→        setAttachedIsImage(false)
   451→      }
   452→      reader.readAsText(file)
   453→    }
   454→    e.target.value = ''
   455→  }

   463→  // ── Feature #2: Chat Export ──
   464→  async function exportChat(format) {
   465→    if (messages.length === 0) return
   466→    const now = new Date()
   467→    const dateStr = now.toISOString().split('T')[0]
   468→    const timeStr = now.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })
   469→    const currentModel = MODELS.find(m => m.id === model)
   470→    const modelName = currentModel?.name || model
   471→
   472→    try {
   473→      const { exportChatAsPDF, exportChatAsDOCX, downloadBlob } = await import('@/lib/fileExportHeavy')
   474→      const { exportChatAsHTML } = await import('@/lib/fileExportClient')
   475→
   476→      if (format === 'pdf') {
   477→        const blob = await exportChatAsPDF(messages, modelName)
   478→        downloadBlob(blob, `kivora-chat-${dateStr}.pdf`)
   479→      } else if (format === 'docx') {
   480→        const blob = await exportChatAsDOCX(messages, modelName)
   481→        downloadBlob(blob, `kivora-chat-${dateStr}.docx`)
   482→      } else if (format === 'html') {
   483→        const blob = exportChatAsHTML(messages, modelName)
   484→        downloadBlob(blob, `kivora-chat-${dateStr}.html`)
   485→      } else {
   486→        // MD and TXT — no external deps needed
   487→        let content = ''
   488→        if (format === 'md') {
   489→          content = `# Kivora AI Chat Export\n\n`
   490→          content += `**Date:** ${dateStr} ${timeStr}\n`
   491→          content += `**Model:** ${modelName}\n\n`
   492→          content += `---\n\n`
   493→          messages.forEach(msg => {
   494→            if (msg.role === 'user') {
   495→              content += `### User\n\n${msg.content}\n\n`
   496→            } else {

   661→  async function checkMicPermission() {
   662→    try {
   663→      const result = await navigator.permissions.query({ name: 'microphone' })
   664→      return result.state // 'granted', 'denied', or 'prompt'
   665→    } catch {
   666→      // permissions.query not supported in some browsers — return null so we fall through to getUserMedia
   667→      return null
   668→    }

   671→  async function toggleListening(e) {
   672→    // Do NOT call e.preventDefault() — it blocks the browser's permission popup
   673→    // by cancelling the "user gesture" that the browser requires.
   674→    if (e) { e.stopPropagation() }
   675→    if (typeof window === 'undefined') return
   676→    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition
   677→
   678→    if (!SpeechRecognition) {
   679→      showVoiceToast('error', t('chat.voice_not_supported') || 'Voice input is not supported in this browser. Please use Chrome or Edge.')
   680→      return
   681→    }
   682→
   683→    if (isListening) {
   684→      if (recognitionRef.current) {
   685→        recognitionRef.current.stop()
   686→      }
   687→      setIsListening(false)
   688→      return
   689→    }
   690→
   691→    // Step 1: Check if permission is already permanently denied.
   692→    // If so, getUserMedia will silently reject without showing a popup,
   693→    // so we must show a modal with manual reset instructions instead.
   694→    const permState = await checkMicPermission()
   695→    if (permState === 'denied') {
   696→      setMicPermModal(true)
   697→      return
   698→    }
   699→
   700→    // Step 2: If permission is 'prompt' (never asked) or 'granted', proceed.
   701→    // For 'prompt': getUserMedia will trigger the browser's permission popup.
   702→    // For 'granted': getUserMedia will succeed immediately.
   703→    // If permissions.query is unsupported (permState === null), try getUserMedia anyway.
   704→    if (permState === 'prompt' || permState === null) {
   705→      try {
   706→        const stream = await navigator.mediaDevices.getUserMedia({ audio: true })
   707→        // Release the stream immediately — we only needed the permission grant
   708→        stream.getTracks().forEach(track => track.stop())
   709→      } catch (err) {
   710→        if (err.name === 'NotAllowedError' || err.name === 'PermissionDeniedError') {
   711→          // User just clicked "Block" on the popup, or it was auto-denied
   712→          setMicPermModal(true)
   713→        } else if (err.name === 'NotFoundError') {
   714→          showVoiceToast('error', t('chat.voice_no_mic') || 'No microphone found. Please connect a microphone.')
   715→        } else {
   716→          showVoiceToast('error', t('chat.voice_error') || 'Voice input error. Please try again.')
   717→        }
   718→        return
   719→      }
   720→    }
   721→
   722→    // Step 3: Permission is granted — start SpeechRecognition
   723→    const recognition = new SpeechRecognition()
   724→    recognition.continuous = true
   725→    recognition.interimResults = true
   726→    recognition.lang = 'en-US'
   727→
   728→    let finalTranscript = ''
   729→
   730→    recognition.onresult = (event) => {
   731→      for (let i = event.resultIndex; i < event.results.length; i++) {
   732→        const transcript = event.results[i][0].transcript
   733→        if (event.results[i].isFinal) {
   734→          finalTranscript += transcript
   735→        }
   736→      }
   737→      if (finalTranscript) {
   738→        setInput(prev => prev ? prev + ' ' + finalTranscript : finalTranscript)
   739→        finalTranscript = ''
   740→        if (textareaRef.current) {
   741→          setTimeout(() => autoResize(textareaRef.current), 0)
   742→        }
   743→      }
   744→    }
   745→
   746→    recognition.onerror = (event) => {
   747→      setIsListening(false)
   748→      if (event.error === 'not-allowed') {
   749→        setMicPermModal(true)
   750→      } else if (event.error === 'no-speech') {
   751→        showVoiceToast('info', t('chat.voice_no_speech') || 'No speech detected. Please try again.')
   752→      } else if (event.error === 'audio-capture') {
   753→        showVoiceToast('error', t('chat.voice_no_mic') || 'No microphone found. Please connect a microphone.')
   754→      } else {
   755→        showVoiceToast('error', t('chat.voice_error') || 'Voice input error. Please try again.')
   756→      }
   757→    }
   758→
   759→    recognition.onend = () => {
   760→      setIsListening(false)
   761→    }
   762→
   763→    recognitionRef.current = recognition
   764→    setIsListening(true)
   765→    try {
   766→      recognition.start()
   767→      showVoiceToast('info', t('chat.voice_listening') || 'Listening... Speak now.')
   768→    } catch (e) {
   769→      setIsListening(false)
   770→      showVoiceToast('error', t('chat.voice_error') || 'Voice input error. Please try again.')
   771→    }
   772→  }
